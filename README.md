# 不会建模的程序员不是好的逆向工程师

```assembly
section .data
    message db '不建模哪有不疯的强撑罢了', 10, 0

section .text
    global _start

_start:
    jmp .身为二次元你不建模?

.身为二次元你不建模?:
    mov 理智, 疯狂     ; 不建模哪有不疯的
    cmp 理智, 0
    jnz .强撑          ; 强撑罢了

.强撑:
    push 精神
    call _print
    jmp .身为二次元你不建模?

.精神:
    db '不建模哪有不疯的强撑罢了', 0

_print:
    mov ecx, message_length
    mov ebx, message
    mov eax, 4
    mov edx, ecx
    mov ecx, ebx
    mov ebx, 1
    int 0x80
    ret

section .data
message_length equ $ - message
```
